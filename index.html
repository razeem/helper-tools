Git PR format and skeleton

DEV
git checkout development && git pull && git checkout -b /-/-develop
git commit -m "[-][DEV]: "
git push -u origin /-/-develop
STAGE
git checkout stage && git pull && git checkout -b /-/-stage && git cherry-pick && git commit --amend -m "[-][STAGE]: "
git push -u origin /-/-stage
MASTER
git checkout clean && git pull && git checkout -b /-/-master && git cherry-pick && git commit --amend -m "[-][CLEAN]: "
git push -u origin /-/-master
